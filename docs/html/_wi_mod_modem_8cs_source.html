81  /// <summary>
82  /// Transmits the passed <paramref name="message"/> to the physical modem.
83  /// </summary>
84  /// <param name="message">the message to be sent.</param>
85  /// <returns>A <see cref="Task"/> representing the asynchronous operation.</returns>
86  public async Task TransmitMessage(TxHciMessage message)
87  {
88  // Retreieve the payload from the message.
89  List<byte> bytes = message.Payload ?? new List<byte>();
90 
91  // Add recipient bytes to the start of the payload.
92  bytes.Insert(0, (byte)message.EndpointIdentifier);
93  bytes.Insert(1, message.MessageIdentifier);
94 
95  // Encode the frame as SLIP.
96  bytes = bytes.EncodeSlipFrame(appendCrc: true).ToList();
97  _logger.LogDebug("Transmitting {bytes}", bytes.ToHexString());
98  await _serialConnection.BaseStream.WriteAsync(bytes.ToArray(), 0, bytes.Count);
99  }

103  {
104  // TODO: Implement some kind of streamreader, for chunking in C0, discarding any and all bytes which are not part of a C0....C0 packet.
105  // When sending Reset, the device sends some weird ass 0xFF - So make sure to deal with invalid bytes!!!
106  var data = new byte[_serialConnection.BytesToRead]; // Get rid of this buggy property. Does not gurantee to tell you how many bytes we want.
107  _serialConnection.Read(data, 0, data.Length);
108  _logger.LogTrace("Received {bytes}", data.ToHexString());
109 
110  IList<byte> decoded;
111  try
112  {
113  _logger.LogDebug("Attempting to unslip {bytes}", data.ToHexString());
114  decoded = data.StripSlipFrame();
115  }
116  catch (Exception ex)
117  {
118  _logger.LogError(ex, "Failed unslipping bytes {bytes}. Reason: {reason}", data.ToHexString(), ex.Message);
119  return;
120  }
121 
122  RxHciMessage message;
123  try
124  {
125  _logger.LogDebug("Attempting to marshal {bytes} into an RX message.", decoded.ToHexString());
126  message = RxHciMessageFactory.Create(decoded);
127  }
128  catch (Exception ex)
129  {
130  _logger.LogError(ex, "Failed marshalling bytes {bytes} into a RX message. Reason: {reason}", decoded.ToHexString(), ex.Message);
131  return;
132  }
133 
134  _logger.LogDebug("Marshalling successful, received: {message}", message.ToString());
135 
136  _onData(message).Wait();
137  }
138  }
