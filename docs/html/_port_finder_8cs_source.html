55  private string FindConnectedPort()
56  {
57  _logger.LogTrace("Starting search for virtual COM device with name: {name}", _deviceName);
58  string foundPort = null;
59  try
60  {
61  var search = new ManagementObjectSearcher(
62  "root\\CIMV2",
63  "SELECT * FROM Win32_PnPEntity WHERE ClassGuid=\"{4d36e978-e325-11ce-bfc1-08002be10318}\"");
64 
65  foreach (ManagementBaseObject mgmtObj in search.Get())
66  {
67  var queryObj = mgmtObj as ManagementObject;
68  var name = queryObj[_mgmtObjQuery] as string;
69  if (name.Contains(_deviceName))
70  {
71  var port = Regex.Match(name, _portRegex).Groups[1].Value;
72  foundPort = port;
73  break;
74  }
75  }
76  }
77  catch (Exception e)
78  {
79  _logger.LogError(e, "Error whilst looking for virtual COM device {name}", _deviceName);
80  }
81 
82  return foundPort;
83  }
84 
85  private SerialPort GetConnectedSerialPort()
86  {
87  var port = FindConnectedPort();
88  return port != null ? new SerialPort(FindConnectedPort()) : null;
89  }
