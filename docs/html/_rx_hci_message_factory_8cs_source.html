32  private static void RegisterAllRxHciMessageTypes()
33  {
34  Type rxHciBaseType = typeof(RxHciMessage);
35  foreach (Assembly domainAssembly in AppDomain.CurrentDomain.GetAssemblies())
36  {
37  foreach (Type type in domainAssembly.GetTypes()
38  .Where(type => type.IsSubclassOf(rxHciBaseType) && !type.IsAbstract))
39  {
40  // this is really ugly, and should properly never be done. I, however cannot think of a more painless way of creating a factory.
41  // Also forces whoever is implementing new messages to have lazy initialization - Which I am fine with.
42  var instance = (RxHciMessage)Activator.CreateInstance(type, new byte[] { });
43  var endpointValue = (byte)type.GetProperty("EndpointIdentifier").GetValue(instance);
44  var messageValue = (byte)type.GetProperty("MessageIdentifier").GetValue(instance);
45  _knownSubTypes.Add((endpointIdentifier: endpointValue, messageIdentifier: messageValue), type);
46  }
