95  /// <summary>
96  /// Encodes as SLIP.
97  /// </summary>
98  /// <param name="bytes">the bytes to be encoded.</param>
99  /// <param name="appendCrc">optional boolean indicating if CRC16 should be appended, before encoding SLIP.</param>
100  /// <returns>the <paramref name="bytes"/>, but SLIP encoded.</returns>
101  public static IList<byte> EncodeSlipFrame(this IList<byte> bytes, bool appendCrc = true)
102  {
103  var result = new List<byte>();
104  if (appendCrc)
105  {
106  bytes.AppendCrc();
107  }
108 
109  foreach (var b in bytes)
110  {
111  if (b == SlipEnd)
112  {
113  result.Add(SlipEscape);
114  result.Add(SlipEscapeEnd);
115  }
116  else if (b == SlipEscape)
117  {
118  result.Add(SlipEscape);
119  result.Add(SlipEscapeEscape);
120  }
121  else
122  {
123  result.Add(b);
124  }
125  }
126 
127  result.Insert(0, SlipEnd);
128  result.Add(SlipEnd);
129 
130  return result;
131  }

133  /// <summary>
134  /// Transforms a hex-formatted string to a byte array.
135  /// </summary>
136  /// <param name="hexString">the string.</param>
137  /// <returns>a byte array of the hex-strings binary value.</returns>
138  public static byte[] HexStringToBytes(this string hexString)
139  {
140  var sanitizedHex = hexString.ToUpper();
141 
142  if (sanitizedHex.Length % 2 == 1)
143  {
144  throw new ArgumentException("The binary key cannot have an odd number of digits");
145  }
146 
147  var arr = new byte[sanitizedHex.Length >> 1];
148 
149  for (var i = 0; i < sanitizedHex.Length >> 1; ++i)
150  {
151  arr[i] = (byte)((GetHexVal(sanitizedHex[i << 1]) << 4) +

158  /// <summary>
159  /// Decodes and strips SLIP.
160  /// </summary>
161  /// <param name="bytes">the bytes to be decoded and stripped.</param>
162  /// <param name="validateCrc">optional boolean indicating if CRC16 should be validated.</param>
163  /// <returns>the decoded frame.</returns>
164  public static IList<byte> StripSlipFrame(this IList<byte> bytes, bool validateCrc = true)
165  {
166  // IList is nice and all, but i'd rather make a copy, than deal with an array might being passed.
167  var list = bytes.ToList();
168 
169  for (var i = 0; i < list.Count; i++)
170  {
171  // Do we have a slip-escape and an escape end? If yes, remove them and replace with a slip-end.
172  // Checks for bounds, so were safe. Could be written prettier - But honestly who cares?
173  if (list[i] == SlipEscape && i + 1 < list.Count && list[i + 1] == SlipEscapeEnd)
174  {
175  list.RemoveAt(i);
176  list.RemoveAt(i);
177  list.Insert(i, SlipEnd);
178  }
179 
180  // Removes escape escape.
181  if (list[i] == SlipEscape && i + 1 < list.Count && list[i + 1] == SlipEscapeEscape)
182  {
183  list.RemoveAt(i + 1);
184  }
185  }
186 
187  if (list[0] == SlipEnd && list[list.Count - 1] == SlipEnd)
188  {
189  list.RemoveAt(0);
190  list.RemoveAt(list.Count - 1);
191  }
192 
193  if (validateCrc)
194  {
195  var crcCorrect = ValidateCrc(list);
196 
197  if (crcCorrect)
198  {
199  list.RemoveAt(list.Count - 1);
200  list.RemoveAt(list.Count - 1);
201  }
202  else
203  {
204  throw new ArgumentException("Frame did not have a valid checksum.");
205  }
206  }
207 
208  return list;
209  }

257  /// <returns>A hex formatted string.</returns>
258  public static string ToHexString(this IList<byte> bytes)
259  {
260  var c = new char[bytes.Count * 2];
261  for (var i = 0; i < bytes.Count; i++)
262  {
263  var b = bytes[i] >> 4;
264  c[i * 2] = (char)(55 + b + (((b - 10) >> 31) & -7));
265  b = bytes[i] & 0xF;

267  }

287  /// <summary>
288  /// Calculates the CRC16 CCITTT.
289  /// </summary>
290  /// <param name="bytes">the bytes.</param>
291  /// <returns>CRC16 CCITTT.</returns>
292  public static ushort CalculateCrc(this IList<byte> bytes)
293  {
294  var crc = InitialCrcValue;
295 
296  foreach (var b in bytes)
297  {
298  crc = (ushort)((crc >> 8) ^ CRC16Table[(crc ^ b) & 0x00FF]);
299  }
300 
301  // WiMod uses ones complement
302  crc = (ushort)~crc;
303  return crc;
304  }
